# Phase 8a — Δv planner web page implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the `/dv` page (form + result + JSON + HTMX partial + shareable URL) at parity with `ksp dv`. No save-as-plan, no plans nav. Foundation for Phase 8 sub-phases.

**Architecture:** New `src/ksp_planner/web/` package alongside `cli.py`. Routes parse Pydantic schemas, call `dv_map.plan_trip` / `plan_round_trip` (untouched), and render Jinja templates. One handler returns three shapes (HTML / HTMX partial / JSON) by content-negotiation on `Accept` and `HX-Request` headers. Design tokens live in `theme.css`.

**Tech Stack:** FastAPI 0.115+, Pydantic 2, Jinja2, HTMX (vendored), uvicorn[standard], pytest + FastAPI `TestClient`. No npm, no build step.

**Reference spec:** [`docs/superpowers/specs/2026-04-24-phase-8-web-ui-design.md`](../specs/2026-04-24-phase-8-web-ui-design.md).

```python
from ksp_planner import db as dblib


def _db_path() -> Path:

    """Yield a read-only connection. Closed after the request."""

        yield conn
```

```python
"""Pydantic request/response schemas for the web layer.

Same models validate Form data and JSON requests. Response models adapt
the pure calculator dataclasses (`TripPlan`, `Edge`, `Stop`) for serialisation.
"""

from __future__ import annotations

from typing import Literal

from pydantic import BaseModel, ConfigDict, Field

from ksp_planner.dv_map import TripPlan


class StopInput(BaseModel):
    model_config = ConfigDict(extra="forbid")
    body: str = Field(..., min_length=1)
    action: Literal["land", "orbit", "flyby"] = "orbit"


class DvRequest(BaseModel):
    model_config = ConfigDict(extra="forbid", populate_by_name=True)
    from_: str = Field(..., alias="from", min_length=1)
    to: str = Field(..., min_length=1)
    via: list[StopInput] = Field(default_factory=list)
    round_trip: bool = False
    aerobrake: bool = True
    margin_pct: float = Field(5.0, ge=0, le=100)


class LegOut(BaseModel):
    from_slug: str
    to_slug: str
    dv_m_s: float
    can_aerobrake: bool


class StopOut(BaseModel):
    """Intermediate (non-endpoint) stop annotation for the renderer."""
    slug: str
    action: Literal["land", "orbit", "flyby"]
    after_leg_idx: int  # render this annotation after legs[after_leg_idx]


class DvResponse(BaseModel):
    from_slug: str
    to_slug: str
    round_trip: bool
    aerobrake: bool
    margin_pct: float
    legs: list[LegOut]
    stops: list[StopOut]
    total_raw: float
    total_aerobraked: float
    total_planned: float
    total_aerobraked_planned: float
    equivalent_cli: str

    @classmethod
    def from_trip(cls, trip: TripPlan, req: DvRequest, equiv_cli: str) -> "DvResponse":
        # Flatten legs: trip.legs is list[list[Edge]]; we render per-edge.
        # Insert StopOut annotations between legs[i] and legs[i+1] for intermediate stops.
        flat_legs: list[LegOut] = []
        stop_annotations: list[StopOut] = []
        for i, leg in enumerate(trip.legs):
            for edge in leg:
                flat_legs.append(
                    LegOut(
                        from_slug=edge.from_slug,
                        to_slug=edge.to_slug,
                        dv_m_s=edge.dv_m_s,
                        can_aerobrake=edge.can_aerobrake,
                    )
                )
            # After this leg's edges, if this isn't the last leg, the next stop
            # is intermediate (i.e. trip.stops[i+1]).
            if i < len(trip.legs) - 1:
                next_stop = trip.stops[i + 1]
                stop_annotations.append(
                    StopOut(
                        slug=next_stop.slug,
                        action=next_stop.action,
                        after_leg_idx=len(flat_legs) - 1,
                    )
                )
        return cls(
            from_slug=req.from_,
            to_slug=req.to,
            round_trip=req.round_trip,
            aerobrake=trip.aerobrake,
            margin_pct=trip.margin_pct,
            legs=flat_legs,
            stops=stop_annotations,
            total_raw=trip.total_raw,
            total_aerobraked=trip.total_aerobraked,
            total_planned=trip.total_planned,
            total_aerobraked_planned=trip.total_aerobraked_planned,
            equivalent_cli=equiv_cli,
        )


def equivalent_cli(req: DvRequest) -> str:
    """Build the CLI invocation that reproduces this request."""
    parts = ["uv run ksp dv", req.from_, req.to]
    for stop in req.via:
        parts.append(f"--via {stop.body}:{stop.action}")
    if req.round_trip:
        parts.append("--return")
    if not req.aerobrake:
        parts.append("--no-aerobrake")
    if req.margin_pct != 5.0:
        # Match Typer formatting: integer-like values print without trailing zeros.
        margin_str = f"{req.margin_pct:g}"
        parts.append(f"--margin {margin_str}")
    return " ".join(parts)
```

```python
"""Routes for the Δv planner page (Phase 8a)."""

from __future__ import annotations

import sqlite3
from collections import defaultdict

from fastapi import APIRouter, Depends, Request

from ksp_planner.db import load_dv_graph
from ksp_planner.web.deps import get_db

router = APIRouter()


def _slugs_by_body(conn: sqlite3.Connection) -> dict[str, list[str]]:
    """Group all dv_node slugs by body for <optgroup> rendering."""
    graph = load_dv_graph(conn)
    grouped: dict[str, list[str]] = defaultdict(list)
    for slug, node in graph._nodes.items():  # noqa: SLF001 — _nodes is intra-package
        key = node.body_slug or "(transfer)"
        grouped[key].append(slug)
    for key in grouped:
        grouped[key].sort()
    # Stable order: kerbol-system bodies first, then alphabetical.
    return dict(sorted(grouped.items()))


def _bodies_for_via(conn: sqlite3.Connection) -> list[str]:
    """Bodies eligible as via stops — anything with a body_slug in the dv graph."""
    graph = load_dv_graph(conn)
    bodies = sorted({n.body_slug for n in graph._nodes.values() if n.body_slug})  # noqa: SLF001
    return bodies


@router.get("/dv")
def get_dv(request: Request, conn: sqlite3.Connection = Depends(get_db)):
    from ksp_planner.web.app import templates

    return templates.TemplateResponse(
        request,
        "pages/dv.html",
        {
            "active_nav": "dv",
            "version": "0.8.0a",
            "slugs_by_body": _slugs_by_body(conn),
            "bodies": _bodies_for_via(conn),
            "form": None,       # no prepopulated form
            "response": None,   # no result yet
            "error": None,
        },
    )
```

```python
from fastapi import Form
from typing import Annotated

from ksp_planner.dv_map import Stop, plan_round_trip, plan_trip, resolve_stop
from ksp_planner.web.schemas import DvRequest, DvResponse, StopInput, equivalent_cli


def _build_request_from_form(
    from_: str,
    to: str,
    round_trip: bool,
    aerobrake: bool,
    margin_pct: float,
    via_body: list[str] | None,
    via_action: list[str] | None,
) -> DvRequest:
    via_bodies = via_body or []
    via_actions = via_action or []
    via = [
        StopInput(body=b, action=a)
        for b, a in zip(via_bodies, via_actions, strict=True)
    ]
    return DvRequest(
        **{"from": from_},
        to=to,
        via=via,
        round_trip=round_trip,
        aerobrake=aerobrake,
        margin_pct=margin_pct,
    )


def _compute(conn: sqlite3.Connection, req: DvRequest):
    graph = load_dv_graph(conn)
    stops: list[Stop] = [Stop(req.from_)]
    for v in req.via:
        stops.append(resolve_stop(graph, v.body, v.action))
    stops.append(Stop(req.to))
    planner = plan_round_trip if req.round_trip else plan_trip
    return planner(graph, stops, margin_pct=req.margin_pct, aerobrake=req.aerobrake)


@router.post("/dv")
def post_dv(
    request: Request,
    from_: Annotated[str, Form(alias="from")],
    to: Annotated[str, Form()],
    round_trip: Annotated[bool, Form()] = False,
    aerobrake: Annotated[bool, Form()] = False,
    margin_pct: Annotated[float, Form()] = 5.0,
    via_body: Annotated[list[str] | None, Form()] = None,
    via_action: Annotated[list[str] | None, Form()] = None,
    conn: sqlite3.Connection = Depends(get_db),
):
    from ksp_planner.web.app import templates

    req = _build_request_from_form(
        from_, to, round_trip, aerobrake, margin_pct, via_body, via_action
    )
    trip = _compute(conn, req)
    response = DvResponse.from_trip(trip, req, equivalent_cli(req))

    return templates.TemplateResponse(
        request,
        "pages/dv.html",
        {
            "active_nav": "dv",
            "version": "0.8.0a",
            "slugs_by_body": _slugs_by_body(conn),
            "bodies": _bodies_for_via(conn),
            "form": req,
            "response": response,
            "error": None,
        },
    )
```

> **Note on Form booleans:** HTML checkboxes only POST when checked, and the value is the string `"on"`. FastAPI/Pydantic coerce `"on"` → `True` when annotated `bool`. An *unchecked* checkbox sends nothing, falling back to the `False` default — that's why `aerobrake` defaults to `False` in the Form signature even though the schema default is `True`. The schema's `True` only applies on the JSON path.

```python
@router.post("/dv")
def post_dv(
    request: Request,
    from_: Annotated[str, Form(alias="from")],
    to: Annotated[str, Form()],
    round_trip: Annotated[bool, Form()] = False,
    aerobrake: Annotated[bool, Form()] = False,
    margin_pct: Annotated[float, Form()] = 5.0,
    via_body: Annotated[list[str] | None, Form()] = None,
    via_action: Annotated[list[str] | None, Form()] = None,
    conn: sqlite3.Connection = Depends(get_db),
):
    from ksp_planner.web.app import templates
    from pydantic import ValidationError

    req: DvRequest | None = None
    response: DvResponse | None = None
    error: str | None = None
    status: int = 200

    try:
        req = _build_request_from_form(
            from_, to, round_trip, aerobrake, margin_pct, via_body, via_action
        )
        trip = _compute(conn, req)
        response = DvResponse.from_trip(trip, req, equivalent_cli(req))
    except ValidationError as e:
        # Surface the first error message; the spec calls for in-page flash, not field-level UI.
        first = e.errors()[0]
        error = f"{'.'.join(str(p) for p in first['loc'])}: {first['msg']}"
        status = 400
    except (KeyError, ValueError) as e:
        error = str(e).strip("\"'")
        status = 400

    accept = request.headers.get("accept", "")
    wants_json = "application/json" in accept and "text/html" not in accept

    if wants_json:
        if error is not None:
            return templates.TemplateResponse(  # JSONResponse below
                request, "partials/error.json", {"error": error}, status_code=status
            ) if False else _json_error(error, status)
        return response

    is_hx = request.headers.get("hx-request") == "true"
    template = "partials/dv_result.html" if (is_hx and response is not None) else "pages/dv.html"
    if error is not None and is_hx:
        # On HTMX errors, swap in a flash partial so the form stays visible.
        template = "partials/error_flash.html"

    return templates.TemplateResponse(
        request,
        template,
        {
            "active_nav": "dv",
            "version": "0.8.0a",
            "slugs_by_body": _slugs_by_body(conn),
            "bodies": _bodies_for_via(conn),
            "form": req,
            "response": response,
            "error": error,
        },
        status_code=status,
    )


def _json_error(message: str, status: int):
    from fastapi.responses import JSONResponse
    return JSONResponse({"error": message}, status_code=status)
```
